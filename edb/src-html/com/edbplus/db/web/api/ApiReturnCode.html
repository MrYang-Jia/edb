016package com.edbplus.db.web.api;
017
018/**
019 * @program: coreframework-parent
020 * @description: API数据返回码
021 * @author: 杨志佳
022 * @create: 2021-01-12 11:27
023 **/
024public enum ApiReturnCode {
025
026    SUCCESS (0,"成功"),
027    FAIL (500,"系统异常"),
028    ;
029
030
031    private Integer code;
032    private String desc;
033
034    ApiReturnCode(Integer code, String desc) {
035        this.code = code;
036        this.desc = desc;
037    }
038
039    public Integer getCode() {
040        return code;
041    }
042
043    public void setCode(Integer code) {
044        this.code = code;
045    }
046
047    public String getDesc() {
048        return desc;
049    }
050
051    public void setDesc(String desc) {
052        this.desc = desc;
053    }
054
055}
