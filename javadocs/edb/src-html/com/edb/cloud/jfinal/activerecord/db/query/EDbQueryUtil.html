019    /**
020     * 通过jpa过滤器加载生成sql和入参
021     * @param eDbFilter
022     * @param andSqlStr
023     * @param paramsList
024     */
025    public static void loadFilter(EDbFilter eDbFilter, StringBuffer andSqlStr, List<Object> paramsList){
026        // 等于
027        if(eDbFilter.getOperator() == EDbFilter.Operator.eq){
028            andSqlStr.append(" = ? ");
029            paramsList.add(eDbFilter.getValue());
030
031        }
032
033        // 不等于
034        if(eDbFilter.getOperator() == EDbFilter.Operator.ne){
035            andSqlStr.append(" <> ? ");
036            paramsList.add(eDbFilter.getValue());
037
038        }
039
040        // 大于
041        if(eDbFilter.getOperator() == EDbFilter.Operator.gt ){
042            andSqlStr.append(" > ? ");
043            paramsList.add(eDbFilter.getValue());
044
045        }
046
047        // 大于等于
048        if(eDbFilter.getOperator() == EDbFilter.Operator.ge  ){
049            andSqlStr.append(" >= ? ");
050            paramsList.add(eDbFilter.getValue());
051
052        }
053
054        // 小于
055        if(eDbFilter.getOperator() == EDbFilter.Operator.lt ){
056            andSqlStr.append(" < ? ");
057            paramsList.add(eDbFilter.getValue());
058
059        }
060
061        // 小于等于
062        if(eDbFilter.getOperator() == EDbFilter.Operator.le ){
063            andSqlStr.append(" <= ? ");
064            paramsList.add(eDbFilter.getValue());
065
066        }
067
068        // 模糊匹配
069        if(eDbFilter.getOperator() == EDbFilter.Operator.like  ){
070            andSqlStr.append(" like ? ");
071            paramsList.add(eDbFilter.getValue());
072
073        }
074
075        // in
076        if(eDbFilter.getOperator() == EDbFilter.Operator.in){
077            andSqlStr.append(" in (");
078            filterArrayFun(eDbFilter,andSqlStr,paramsList);
079            andSqlStr.append(")");
080        }
081
082        // not in
083        if(eDbFilter.getOperator() == EDbFilter.Operator.notIn){
084            andSqlStr.append(" not in (");
085            filterArrayFun(eDbFilter,andSqlStr,paramsList);
086            andSqlStr.append(")");
087        }
088
089        // is not null
090        if(eDbFilter.getOperator() == EDbFilter.Operator.isNotNull){
091            andSqlStr.append(" is not null");
092        }
093
094        // is null
095        if(eDbFilter.getOperator() == EDbFilter.Operator.isNull){
096            andSqlStr.append(" is null");
097        }
098    }

100    /**
101     * 处理 fileter 数组逻辑部分
102     * @param EDbFilter
103     * @param andSqlStr
104     * @param paramsList
105     */
106    public static void filterArrayFun(EDbFilter EDbFilter, StringBuffer andSqlStr, List<Object> paramsList){
107        if (EDbFilter.getValue() instanceof Object[] ||
108                EDbFilter.getValue() instanceof String[] ||
109                EDbFilter.getValue() instanceof Integer[] ||
110                EDbFilter.getValue() instanceof Long[] ||
111                EDbFilter.getValue() instanceof ArrayList ) {
112            Object[] values = null;
113            if(EDbFilter.getValue() instanceof ArrayList){
114                values = ((List<Object>) EDbFilter.getValue()).toArray();
115            }else{
116                values = (Object[]) EDbFilter.getValue();
117            }
118            for(int j=0; j<values.length; j++) {
119                andSqlStr.append("?");
120                if(j < values.length - 1) {
121                    andSqlStr.append(",");
122                }
123                paramsList.add(values[j]);
124            }
125        }
126    }

161    public static SqlPara getSqlParaForJpaQuery(String tableName, EDbQuery queryParams){
162
163
164        if(queryParams == null){
165            queryParams = new EDbQuery();
166        }
167
168        //
169        List<Object> paramsList = new ArrayList<>();
170        //
171        StringBuffer andSqlStr =  new StringBuffer("");
172        // 首次拼接前用 1=1 ，以便后续的对象可以 and 拼接
173        andSqlStr.append(" 1=1 ");
174
175        if (queryParams.getAndEDbFilters().size()>0){
176            andSqlStr.append(" and ");
177            // 处理当前 and filter 部分 和 or fillter 部分
178            baseQueryFun(queryParams,andSqlStr,paramsList);
179        }
180
181        // and ( filters ) 部分
182        if(queryParams.andCom().getAndEDbFilters().size()>0 ) {
183            andSqlStr.append(" and (  ");
184            baseQueryFun(queryParams.andCom(), andSqlStr, paramsList);
185            andSqlStr.append(" )");
186        }
187
188        // or ( filters ) 部分
189        if(queryParams.orCom().getAndEDbFilters().size()>0 ){
190            andSqlStr.append(" or (  ");
191            baseQueryFun( queryParams.orCom(),andSqlStr,paramsList);
192            andSqlStr.append(" )");
193        }
194
195        // 将 2=2 去掉，是为了保证不会被识别成是注入的代码块
196        String andSql =  andSqlStr.toString().replaceAll("2=2  and","");
197
198
199        StringBuffer orderSql =  new StringBuffer("");
200        Order order = null;
201        // order 部分
202        if(queryParams.getOrders().size() > 0){
203            orderSql.append(" order by ");
204        }
205        for(int i=0;i< queryParams.getOrders().size();i++){
206            //
207            order = (Order) queryParams.getOrders().get(i);
208            //
209            orderSql.append(order.getProperty()).append(" ").append(order.getDirection().name());
210            //
211            if(i < queryParams.getOrders().size() - 1) {
212                orderSql.append(",");
213            }
214        }
215        // sqlPara
216        SqlPara sqlPara = new SqlPara();
217        // select xxx
218        if( queryParams.getFieldsSql() != null){
219            sqlPara.setSql( "select " + queryParams.getFieldsSql() + " from " + tableName + " where  " + andSql + queryParams.getLastSql()  + orderSql );
220        }else{
221            //
222            sqlPara.setSql("select * from "+ tableName + " where  " + andSql + queryParams.getLastSql()   + orderSql );
223        }
224        // 传递参数对象
225        for(Object para:paramsList){
226            sqlPara.addPara(para);
227        }
228        return sqlPara;
229    }

231    /**
232     * 根据jpa单表查询对象返回 jdbcSql 查询对象
233     * @param mClass
234     * @param queryParams
235     * @return
236     */
237    public static SqlPara getSqlParaForJpaQuery(Class<?> mClass, EDbQuery queryParams){
238        // 获取表注解
239        Table table = AnnotationUtil.getAnnotation( mClass , Table.class);
240        if(table == null){
241            throw new RuntimeException("@Table is not find");
242        }
243        return getSqlParaForJpaQuery(table.name(),queryParams);
244    }
