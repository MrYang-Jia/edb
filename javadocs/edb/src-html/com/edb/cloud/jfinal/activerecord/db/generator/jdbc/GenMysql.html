012    /**
013     * 识别用 , 分割的多表语句，并改写语句
014     * @param tableName
015     * @return
016     */
017    public static String getTablesSql(String tableName){
018
019        // 如果已经包含了单引号，则不需要再进行一次改写
020        if(tableName.contains("'")){
021            return tableName;
022        }
023
024        String[] tbs = tableName.split(",");
025        StringBuilder sql = new StringBuilder("");
026        for(String tb : tbs){
027            // 前面有字段时，添加逗号即可
028            if(sql.length() > 0){
029                sql.append(",");
030            }
031            sql.append("'").append(tb).append("'");
032        }
033        return sql.toString();
034    }

036    /**
037     * 返回表对象信息Sql
038     * @param tableName
039     * @return
040     */
041    public static String getTableInfoSql(String tableName){
042        StringBuilder sql = new StringBuilder();
043        sql.append(" SELECT ");
044        sql.append("    table_name tableName ");
045        sql.append("    ,ENGINE ");
046        sql.append("    ,table_comment tableComment ");
047        sql.append("    ,create_time createTime ");
048        sql.append(" FROM ");
049        sql.append("    information_schema. TABLES ");
050        sql.append(" WHERE ");
051        sql.append("    table_schema = (SELECT DATABASE()) ");
052        // table表名称
053        if(tableName != null && tableName.length() > 0){
054            if(tableName.contains(",")){
055                sql.append(" AND table_name in (").append(getTablesSql(tableName)).append(") ");
056            }else{
057                sql.append(" AND table_name = '").append(tableName).append("' ");
058            }
059        }
060        sql.append(" order by table_name ");
061        return sql.toString();
062    }

084    /**
085     * 根据传入的
086     * @param tableName
087     * @param isOnlyAutoIncrement
088     * @return
089     */
090    public static String getTableColumnsSql(String tableName,boolean isOnlyAutoIncrement ){
091        StringBuilder sql = new StringBuilder();
092        sql.append(" SELECT ");
093        // 添加顺序
094        sql.append(" ordinal_position ordinalPosition, ");
095        // 添加表名字段
096        sql.append(" table_name tableName, ");
097        sql.append("    column_name columnName, ");
098        sql.append("    COLUMN_DEFAULT columnDefault, ");
099        sql.append("    data_type dataType, ");
100        sql.append("    column_comment columnComment, ");
101        sql.append("    column_key columnKey, ");
102        sql.append("    extra, ");
103        sql.append("    ( ");
104        sql.append("            CASE ");
105        sql.append("            WHEN IS_NULLABLE = 'YES' THEN ");
106        sql.append("                    '1' ");
107        sql.append("            ELSE ");
108        sql.append("                    '0' ");
109        sql.append("            END ");
110        sql.append("    ) isN, ");
111        sql.append("    SUBSTRING( ");
112        sql.append("            column_type, ");
113        sql.append("            INSTR(column_type, '(') + 1, ");
114        sql.append("            INSTR(column_type, ')') - INSTR(column_type, '(') - 1 ");
115        sql.append("    ) maxL ");
116        sql.append(" FROM ");
117        sql.append("    information_schema. COLUMNS ");
118        sql.append(" WHERE ");
119        // 如果没有传入表单，则全部查询
120        if( tableName!=null && tableName.length()>0 ){
121            if(tableName.contains(",")){
122                sql.append("  table_name in (").append(getTablesSql(tableName)).append(") ");
123            }else{
124                sql.append("  table_name = '").append(tableName).append("' ");
125            }
126        }else
127        {
128            sql.append("  1=1 ");
129        }
130        // 查询指定的当前库
131        sql.append(" AND table_schema = (SELECT DATABASE()) ");
132        // 只获取自增键值的语句
133        if(isOnlyAutoIncrement){
134            // 区分正则
135            sql.append(" and extra = 'auto_increment' ");
136        }
137        sql.append(" ORDER BY ");
138        // 根据表名和字段顺序排序
139        sql.append("    table_name,ordinal_position ");
140        return sql.toString();
141    }

170    /**
171     * 返回表分布式键 -- 基于唯一值的字段返回
172     * @param tableName
173     * @return
174     */
175    public static String getIndexForDistributed(String tableName){
176        StringBuffer sql = new StringBuffer("");
177        sql.append(" select  table_schema tableSchema, ");
178        sql.append(" table_name tableName, ");
179        sql.append(" non_unique notUnique, ");
180        // 根据字段名进行组合，其中重复的字段剔除
181        sql.append(" group_concat(distinct column_name) columnName ");
182
183        sql.append(" from INFORMATION_SCHEMA.STATISTICS");
184
185        sql.append(" where 1=1 ");
186        // 当前数据库
187        sql.append(" AND table_schema = ( SELECT DATABASE ( ) ) ");
188        // 只获取唯一值(含主键)
189        sql.append(" and non_unique = 0 ");
190        // 如果table有值的话
191        if( tableName != null ){
192            if(tableName.contains(",")){
193                sql.append(" and table_name in (").append(getTablesSql(tableName)).append(") ");
194            }else{
195                sql.append(" and table_name = '").append(tableName).append("' ");
196            }
197
198        }
199        // 根据数据库，表名，非唯一值 统计
200        sql.append(" group by table_schema,table_name,non_unique ");
201        // 根据表名顺序排序
202        sql.append(" order by table_name ");
203
204        return sql.toString();
205    }

215    /**
216     * 返回表的所有索引,并将索引字段组合起来
217     * @param tableName
218     * @return
219     */
220    public static String getIndex(String tableName){
221        StringBuffer sql = new StringBuffer("");
222        sql.append(" select  table_schema tableSchema, ");
223        sql.append(" table_name tableName, ");
224        sql.append(" non_unique notUnique, ");
225        // 索引名称
226        sql.append(" index_name indexName, ");
227        // 根据字段名进行组合，其中重复的字段剔除
228        sql.append(" group_concat(column_name) columnName ");
229
230        sql.append(" from INFORMATION_SCHEMA.STATISTICS");
231
232        sql.append(" where 1=1 ");
233        // 当前数据库
234        sql.append(" AND table_schema = ( SELECT DATABASE ( ) ) ");
235        // 如果table有值的话
236        if( tableName != null ){
237            if( tableName.contains(",") ){
238                sql.append(" and table_name in (").append(getTablesSql(tableName)).append(") ");
239            }else
240            {
241                sql.append(" and table_name = '").append(tableName).append("' ");
242            }
243        }
244        // 根据数据库，表名，非唯一值，索引名称 统计
245        sql.append(" group by table_schema,table_name,non_unique,index_name ");
246        // 根据表名、索引名称(主键优先 -- 因为分布式系统建议唯一键值时，需要优先考虑主键搭配键值建立唯一键值)
247        sql.append(" order by table_name,(case when index_name='PRIMARY' then 0 else 1 end) ");
248
249        return sql.toString();
250    }
