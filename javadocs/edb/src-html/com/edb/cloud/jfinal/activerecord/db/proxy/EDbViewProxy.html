032public class EDbViewProxy implements MethodInterceptor {
033
034
035    //要代理的原始对象
036    private Object oriJpa ;
037    // 数据对象
038    private EDbPro eDbPro;
039
040    // 1- 创建代理对象
041    public <T> T createProcy(T target,EDbPro eDbPro){
042        // 需要判断是否已经是代理对象，如果是的话，无须二次代理
043//        System.out.println("需要被代理的类："+this.getClass().getSimpleName());
044        // 赋予操作对象
045        this.oriJpa = target;
046        // 赋予数据库对象
047        this.eDbPro = eDbPro;
048        //1-Enhancer类是CGLib中的一个字节码增强器，它可以方便的对你想要处理的类进行扩展
049        Enhancer enhancer=new Enhancer();
050        //2-将被代理类HuGe设置成父类
051        enhancer.setSuperclass(this.oriJpa.getClass());
052        //3-设置拦截器
053        enhancer.setCallback(this);
054        //4-动态生成一个代理类
055        Object objProxy = enhancer.create();
056        return (T) objProxy;
057    }
058
059    //2-实现MethodInterceptor的intercept方法
060    public Object intercept(Object obj, Method method, Object[] args, MethodProxy proxy) throws Throwable {
061//        System.out.println("before: " + method);
062        //调用proxy.invoke()方法，会报java.lang.StackOverflowError错误，原因是invoke()内部会一直被反复调用
063        //Object object = proxy.invoke(obj, args);
064//        Object object = proxy.invokeSuper(obj, args);
065        Object object = null;
066        // 获取调用方法的返回对象
067        Type returnType = method.getAnnotatedReturnType().getType();
068        // 如果返回类型有值，才进行关系扩展
069        if(returnType!= null ){
070            // 获取所有字段 -- 包含
071            List<FieldAndView> fieldViews = JpaAnnotationUtil.getFieldViews(oriJpa.getClass());
072            Class<?> entityClass = null;
073            for(FieldAndView fieldAndView :fieldViews) {
074                // 排除掉非引用注解字段的get方法
075                if (method != null && !method.getName().equals(JpaAnnotationUtil.getFieldMethod(fieldAndView.getField(), oriJpa.getClass()).getName())) {
076                    continue;
077                }
078                // 将当前对象的所有属性转换成入参对象
079                Map<String,Object> map = new HashMap<>();
080                // 原来只转一层map的时候，bean对象无法使用  map[field] 的语法  //BeanUtil.beanToMap(oriJpa);
081                // 将对象先转成json字符串，再转map对象，这样子对象里的对象，就都是map，否则enjoy解析器，基于bean的内部解析时，无法使用类似 map[field] 的语法
082                map = JSONUtil.toBean(JSONUtil.toJsonStr(oriJpa),map.getClass());
083                //
084                SqlPara sqlPara = eDbPro.getSqlPara(fieldAndView.getEDbView().name(), map);
085                if(sqlPara==null){
086                    throw new RuntimeException("未加载sql视图:" + fieldAndView.getEDbView().name());
087                }
088                if(returnType instanceof ParameterizedType){
089                    // List
090                    entityClass = (Class<?>)((ParameterizedType) returnType).getActualTypeArguments()[0];
091                    object = eDbPro.find(entityClass,sqlPara);
092
093                }else{
094                    entityClass = Class.forName(returnType.getTypeName());
095                    object = eDbPro.findFirst(entityClass,sqlPara);
096                }
097                // 字段赋值 -- 反射赋值会比较消耗毫秒数
098                ReflectUtil.setFieldValue(oriJpa, fieldAndView.getField(), object);
099                // 跳出循环
100                break;
101            }
102        }
103
104        // 如果不是rel对象，则返回原属性方法值
105        if(object == null){
106            object = proxy.invokeSuper(obj, args);
107        }
108        return object;
109    }
