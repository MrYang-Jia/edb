061    /**
062     * 通过map集合组合条件，返回查询对象
063     * @param mClass
064     * @param whereMap
065     * @return
066     */
067    public static EDbQuery getQueryForFilter(Class<?> mClass, Map<String,Object> whereMap){
068        //
069        EDbQuery eDbQuery = new EDbQuery();
070        // 获取驼峰与数据库字段映射集
071        Map<String,String> jpaDbColumns  = JpaKit.getJpaDbColumns(mClass);
072        // 分割单词
073        String filterEc = "";
074        // 字段名
075        String fieldName = "";
076        // 获取排序元素
077        if(whereMap.containsKey("field")){
078            // 排序字段
079            String field = (String) whereMap.get("field");
080            String order = "desc";
081            // 判断是否有排序的模式字段
082            if(whereMap.containsKey("order")){
083                String newOrder = (String) whereMap.get("order");
084                if(!StringUtils.isEmpty(newOrder)){
085                    order = newOrder;
086                }
087            }
088            // 添加排序字段
089            if(order.toLowerCase().equals("desc")){
090                eDbQuery.orderDESC(jpaDbColumns.get(field));
091            }else{
092                eDbQuery.orderASC(jpaDbColumns.get(field));
093            }
094        }
095
096
097        // 获取当前组合元素
098        for (Map.Entry<String, Object> mapData : whereMap.entrySet()) {
099            // 拼接组合   and x1 and x2
100            if (mapData.getKey().contains(EDbFilterKit.filterPre)){
101                // 默认是 s_eq_驼峰字段名
102                filterEc = mapData.getKey().split(EDbFilterKit.splitStr)[1];
103                fieldName = mapData.getKey().split(EDbFilterKit.splitStr)[2];
104                // 加载对象
105                loadEDbQueryForAnd(eDbQuery,filterEc,fieldName,mapData.getValue(),jpaDbColumns);
106            }
107        }
108
109
110        // andCom 不为 null 的情况
111        if( whereMap.get("andCom") != null){
112            //
113            Map<String,Object> andCom = (Map<String, Object>) whereMap.get("andCom");
114            //
115            //andCom = JpaKit.toDbColumnMap(TraVehicleSource.class,andCom);
116            // for循环匹配key值
117            for (Map.Entry<String, Object> mapData : andCom.entrySet()) {
118                // 拼接组合 and (   x1 and x2   )
119                if (mapData.getKey().contains(EDbFilterKit.filterPre)){
120                    // 默认是 s_eq_驼峰字段名
121                    filterEc = mapData.getKey().split(EDbFilterKit.splitStr)[1];
122                    fieldName = mapData.getKey().split(EDbFilterKit.splitStr)[2];
123                    // 加载对象
124                    loadEDbQueryForAnd(eDbQuery.andCom(),filterEc,fieldName,mapData.getValue(),jpaDbColumns);
125                }
126            }
127
128        }
129
130        // orCom 不为 null 的情况
131        if( whereMap.get("orCom") != null){
132            //
133            Map<String,Object> orCom = (Map<String, Object>) whereMap.get("orCom");
134            for (Map.Entry<String, Object> mapData : orCom.entrySet()) {
135                // 拼接组合 or (   x1 and x2   )
136                if (mapData.getKey().contains(EDbFilterKit.filterPre)) {
137                    // 默认是 s_eq_驼峰字段名
138                    filterEc = mapData.getKey().split(EDbFilterKit.splitStr)[1];
139                    fieldName = mapData.getKey().split(EDbFilterKit.splitStr)[2];
140                    // 加载对象
141                    loadEDbQueryForAnd(eDbQuery.orCom(), filterEc, fieldName, mapData.getValue(), jpaDbColumns);
142                }
143            }
144        }
145
146        return eDbQuery;
147    }

149    /**
150     * 拼接组合sql信息
151     * @param baseQuery
152     * @param filterEc
153     * @param fieldName
154     * @param value
155     * @param jpaDbColumns
156     */
157    public static void loadEDbQueryForAnd(EDbBaseQuery baseQuery,String filterEc,String fieldName,Object value,Map<String,String> jpaDbColumns ){
158        if(value!=null){
159            switch(filterEc){
160                // 相等
161                case EDbFilterKit.eq :
162                    baseQuery.and(new EDbFilter(jpaDbColumns.get(fieldName), EDbFilter.Operator.eq,value));
163                    break;
164                // 不相等
165                case EDbFilterKit.ne :
166                    baseQuery.and(new EDbFilter(jpaDbColumns.get(fieldName), EDbFilter.Operator.ne,value));
167                    break;
168                // 大于
169                case EDbFilterKit.gt :
170                    baseQuery.and(new EDbFilter(jpaDbColumns.get(fieldName), EDbFilter.Operator.gt,value));
171                    break;
172                // 大于等于
173                case EDbFilterKit.ge :
174                    baseQuery.and(new EDbFilter(jpaDbColumns.get(fieldName), EDbFilter.Operator.ge,value));
175                    break;
176                // 小于
177                case EDbFilterKit.lt :
178                    baseQuery.and(new EDbFilter(jpaDbColumns.get(fieldName), EDbFilter.Operator.lt,value));
179                    break;
180                // 小于等于
181                case EDbFilterKit.le :
182                    baseQuery.and(new EDbFilter(jpaDbColumns.get(fieldName), EDbFilter.Operator.le,value));
183                    break;
184                // 包含
185                case EDbFilterKit.in :
186                    //
187                    if(value instanceof String){
188                        // 将字符串转成list
189                        baseQuery.and(new EDbFilter(jpaDbColumns.get(fieldName), EDbFilter.Operator.in, Arrays.asList(((String) value).split(","))));
190                    }else{
191                        baseQuery.and(new EDbFilter(jpaDbColumns.get(fieldName), EDbFilter.Operator.in,value));
192                    }
193                    break;
194                // 不包含
195                case EDbFilterKit.notIn :
196                    //
197                    if(value instanceof String){
198                        // 将字符串转成list
199                        baseQuery.and(new EDbFilter(jpaDbColumns.get(fieldName), EDbFilter.Operator.notIn, Arrays.asList(((String) value).split(","))));
200                    }else{
201                        baseQuery.and(new EDbFilter(jpaDbColumns.get(fieldName), EDbFilter.Operator.notIn,value));
202                    }
203                    break;
204                // is null
205                case EDbFilterKit.isNull :
206                    baseQuery.and(new EDbFilter(jpaDbColumns.get(fieldName), EDbFilter.Operator.isNull,null));
207                    break;
208                // is not null
209                case EDbFilterKit.isNotNull :
210                    baseQuery.and(new EDbFilter(jpaDbColumns.get(fieldName), EDbFilter.Operator.isNotNull,null));
211                    break;
212                // like 相似
213                case EDbFilterKit.like :
214                    baseQuery.and(new EDbFilter(jpaDbColumns.get(fieldName), EDbFilter.Operator.like,value));
215                    break;
216                default :
217                    //其他规则
218            }
219        }
220    }

222    /**
223     * 移除空对象
224     * @param baseQuery
225     */
226    public static void removeEmetyForQuery(EDbBaseQuery baseQuery){
227        if(baseQuery.getAndEDbFilters()!=null){
228            for(EDbFilter eDbFilter:baseQuery.getAndEDbFilters()){
229                if (StringUtils.isEmpty(eDbFilter.getValue())){
230                    baseQuery.getAndEDbFilters().remove(eDbFilter);
231                }
232            }
233        }
234
235    }

237    /**
238     * 去除空指针的对象
239     * @param map
240     * @param <K>
241     * @param <V>
242     * @return
243     */
244    public static <K, V> Map<K, V> removeNullValue(Map<K, V> map) {
245        if (MapUtil.isEmpty(map)) {
246            return map;
247        } else {
248            Iterator iter = map.entrySet().iterator();
249            while(iter.hasNext()) {
250                Map.Entry<K, V> entry = (Map.Entry)iter.next();
251                if (StrKit.isBlank(String.valueOf(entry.getValue()))) {
252                    iter.remove();
253                }
254            }
255            return map;
256        }
257    }
