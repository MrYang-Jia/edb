001package com.edb.cloud.jfinal.activerecord.db.query;
002
003
004import java.io.Serializable;
005
006/**
007 * @program: Order
008 * @description: 排序辅助类
009 * @author: 杨志佳
010 * @create: 2020-04-19 15:05
011 **/
012public class Order implements Serializable {
013
014    private static final long serialVersionUID = -3078342809727773232L;
015
016    /**
017     * 方向
018     */
019    public enum Direction {
020
021        /** 递增 */
022        asc,
023
024        /** 递减 */
025        desc
026    }
027
028    /** 默认方向 */
029    private static final Order.Direction DEFAULT_DIRECTION = Order.Direction.desc;
030
031    /** 属性 */
032    private String property;
033
034    /** 方向 */
035    private Order.Direction direction = DEFAULT_DIRECTION;
036
037    @Override
038    public String toString() {
039        return property+" " + direction.name();
040    }
041
042    /**
043     * 构造方法
044     */
045    public Order() {
046    }
047
048    /**
049     * 构造方法
050     *
051     * @param property
052     *            属性
053     * @param direction
054     *            方向
055     */
056    public Order(String property, Order.Direction direction) {
057        this.property = property;
058        this.direction = direction;
059    }
060
061    /**
062     * 返回递增排序
063     *
064     * @param property
065     *            属性
066     * @return 递增排序
067     */
068    public static Order asc(String property) {
069        return new Order(property, Order.Direction.asc);
070    }
071
072    /**
073     * 返回递减排序
074     *
075     * @param property
076     *            属性
077     * @return 递减排序
078     */
079    public static Order desc(String property) {
080        return new Order(property, Order.Direction.desc);
081    }
082
083    /**
084     * 获取属性
085     *
086     * @return 属性
087     */
088    public String getProperty() {
089        return property;
090    }
091
092    /**
093     * 设置属性
094     *
095     * @param property
096     *            属性
097     */
098    public void setProperty(String property) {
099        this.property = property;
100    }
101
102    /**
103     * 获取方向
104     *
105     * @return 方向
106     */
107    public Order.Direction getDirection() {
108        return direction;
109    }
110
111    /**
112     * 设置方向
113     *
114     * @param direction
115     *            方向
116     */
117    public void setDirection(Order.Direction direction) {
118        this.direction = direction;
119    }
120
121
122}
