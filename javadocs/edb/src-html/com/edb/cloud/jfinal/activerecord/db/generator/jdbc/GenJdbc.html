133    /**
134     * 初始化基于enjoy的数据库连接
135     * @param configName
136     * @param jdbcUrl
137     * @param userName
138     * @param pwd
139     * @param sqlTplList
140     * @param shareSqlTplList
141     * @param filterList
142     * @param driverClass
143     */
144    public static void initForEnjoy(String configName,String jdbcUrl,String userName,String pwd,List<String> sqlTplList,List<String> shareSqlTplList,List<Filter> filterList,String driverClass){
145
146
147        if(driverClass!=null){
148            dp = new DruidPlugin(jdbcUrl, userName, pwd,driverClass);
149        }else {
150            dp = new DruidPlugin(jdbcUrl, userName, pwd);
151        }
152
153        // 最小空闲连接数
154        dp.setMinIdle(minIdle);
155        // 最大活动连接数
156        dp.setMaxActive(maxActive);
157
158        if(configName==null){
159            arp = new ActiveRecordPlugin(dp);
160        }else{
161            arp = new ActiveRecordPlugin(configName,dp);
162        }
163
164        // 定于 pg 的解析器
165        if(jdbcUrl.contains("postgresql")){
166            arp.setDialect(new PostgreSqlDialect());
167        }
168
169        // 打印sql
170        arp.setShowSql(false);
171        // 打印sql -- 交予底层统一打印
172//        GenJdbc.arp.setShowSql(false);
173        // 判断是否存在需要添加的sql模板
174        if(sqlTplList!=null){
175            // 设置模板的读取环境 从class的资源目录底下获取
176            arp.getEngine().setToClassPathSourceFactory();
177            for(String sqlTpl:sqlTplList){
178                arp.addSqlTemplate(sqlTpl);
179            }
180        }
181
182        // 判断共享模板是否存在
183        if(shareSqlTplList!=null){
184            for(String sqlTpl:shareSqlTplList){
185                // 添加共享模板
186                arp.getEngine().addSharedFunction(sqlTpl);
187            }
188        }
189        // 获取 Enjoy 模板
190        Engine engine = arp.getEngine();
191        // 启动对 空格 的压缩，例如sql的输出就会更精简
192        engine.setCompressorOn(' ');
193
194        engine.addSharedObject("JpaKit", new JpaKit());
195
196        // 支持模板热加载，绝大多数生产环境下也建议配置成 true，除非是极端高性能的场景
197//        engine.setDevMode(true);
198
199        // 提高性能 ，但首次加载的时候会比较慢，并发时能提高性能，应该是做了缓存
200        //engine.setFastMode(true);
201
202        dp.start();
203        arp.start();
204
205        // 添加druid过滤器，需要重启
206        addFilterList(filterList, (DruidDataSource) dp.getDataSource());
207        try {
208            ((DruidDataSource) dp.getDataSource()).restart();
209        } catch (SQLException e) {
210            e.printStackTrace();
211        }
212
213    }

239    /**
240     * 添加druid 过滤器
241     * @param filterList
242     * @param ds
243     */
244    public static void addFilterList(List<Filter> filterList, DruidDataSource ds) {
245        if (filterList != null) {
246            List<Filter> targetList = ds.getProxyFilters();
247            Iterator var3 = filterList.iterator();
248
249            while(var3.hasNext()) {
250                Filter add = (Filter)var3.next();
251                boolean found = false;
252                Iterator var6 = targetList.iterator();
253
254                while(var6.hasNext()) {
255                    Filter target = (Filter)var6.next();
256                    if (add.getClass().equals(target.getClass())) {
257                        found = true;
258                        break;
259                    }
260                }
261                if (!found) {
262                    targetList.add(add);
263                }
264            }
265        }
