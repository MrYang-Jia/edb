001package com.edb.cloud.jfinal.activerecord.db.jpa.kit;
002
003import com.edb.cloud.jfinal.activerecord.db.dto.FieldAndColumn;
004import com.edb.cloud.jfinal.activerecord.db.jpa.JpaAnnotationUtil;
005
006import java.util.HashMap;
007import java.util.List;
008import java.util.Map;
009
010public class JpaKit {
011
012    /**
013     * 是否是列表
014     * @param self
015     * @return
016     */
017    public static boolean isList(Object self){
018        if(self instanceof java.util.List){
019            return true;
020        }
021        return false;
022    }
023
024    /**
025     * 是否是map类型
026     * @param self
027     * @return
028     */
029    public static boolean isMap(Object self){
030        if(self instanceof java.util.Map){
031            return true;
032        }
033        return false;
034    }
035
036    /**
037     * 根据jpa对象转数据库字段
038     * @param mClass
039     * @param data -- 从前端获取到驼峰字段数据集合
040     * @return
041     */
042    public static  Map<String,Object> toDbColumnMap(Class<?> mClass, Map<String,Object> data){
043        // 获取对象所有字段的列表信息
044        List<FieldAndColumn> fieldAndColumns =  JpaAnnotationUtil.getCoumns(mClass);
045        Map<String,Object> result =  new HashMap<>();
046        // 重置为数据库字段对象
047        for(FieldAndColumn fieldAndColumn : fieldAndColumns){
048            // 判断是否获取到该驼峰字段信息，如果是则转换成数据库的字段
049            if(data.get(fieldAndColumn.getField().getName())!=null){
050                result.put(fieldAndColumn.getColumn().name(),data.get(fieldAndColumn.getField().getName()));
051            }
052        }
053        return result;
054    }
055
056    /**
057     * 获取jpa数据库对照字段集合
058     * @param mClass
059     * @return
060     */
061    public static Map<String,String> getJpaDbColumns(Class<?> mClass){
062        // 获取对象所有字段的列表信息
063        List<FieldAndColumn> fieldAndColumns =  JpaAnnotationUtil.getCoumns(mClass);
064        Map<String,String> result =  new HashMap<>();
065        // 重置为数据库字段对象
066        for(FieldAndColumn fieldAndColumn : fieldAndColumns){
067            // key = 驼峰字段 ，value = 数据库字段
068            result.put(fieldAndColumn.getField().getName(),fieldAndColumn.getColumn().name());
069        }
070        return result;
071    }
072
073
074}
