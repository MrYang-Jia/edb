001package com.edb.cloud.jfinal.activerecord.db.jpa.task;
002
003import cn.hutool.core.util.ReflectUtil;
004import com.edb.cloud.jfinal.activerecord.db.EDbPro;
005import com.jfinal.plugin.activerecord.DbPro;
006import com.jfinal.plugin.activerecord.SqlPara;
007
008import java.lang.reflect.Field;
009import java.util.concurrent.Callable;
010
011public class JpaRelTask implements Callable<Object> {
012
013    private Class<?> jpaClass;
014    private SqlPara sqlPara;
015    private EDbPro eDbPro;
016    private Object oriJpa;
017    private Field field;
018    private boolean isList = false;
019
020    public JpaRelTask(Object oriJpa,Field oriJpaField,Class<?> jpaClass, SqlPara sqlPara, EDbPro eDbPro,boolean isList){
021        this.jpaClass = jpaClass;
022        this.sqlPara = sqlPara;
023        this.eDbPro = eDbPro;
024        this.oriJpa =oriJpa;
025        this.field = oriJpaField;
026        this.isList = isList;
027    }
028
029    @Override
030    public Object call() throws Exception {
031        Object object = null;
032        if(isList){
033            object = eDbPro.find(jpaClass, sqlPara);
034        }else{
035            object = eDbPro.findFirst(jpaClass, sqlPara);
036        }
037        System.out.println("===运行===");
038        // 字段赋值 -- 反射赋值会比较消耗毫秒数
039        ReflectUtil.setFieldValue(oriJpa, field, object);
040
041        return object;
042    }
043
044}
