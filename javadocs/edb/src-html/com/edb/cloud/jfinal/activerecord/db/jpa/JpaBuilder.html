263     * 比较两个对象，返回实际变更的值
264     * @param oldBean
265     * @param newBean
266     * @return
267     */
268    public static Map<String,Object> contrastObjReturnColumnMap(Object oldBean, Object newBean) {
269        Map<String,Object> updateMap = new HashMap<>();
270        // 通过反射获取类的类类型及字段属性
271        Field[] fields = ReflectUtil.getFields(oldBean.getClass());
272        Object o1 ;
273        Object o2 ;
274        int i = 1;
275        Column column ;
276        for (Field field : fields) {
277            // 排除序列化属性
278            if ("serialVersionUID".equals(field.getName())) {
279                continue;
280            }
281            o1 = ReflectUtil.getFieldValue(oldBean,field);
282            o2 = ReflectUtil.getFieldValue(newBean,field);
283            //
284            if(o1 != o2){
285                //
286                column =  AnnotationUtil.getAnnotation(field, Column.class);
287                if(column != null ){
288                    // 赋予新的变更值
289                    updateMap.put(column.name().toLowerCase(),o2);
290                }
291            }
292        }
293        // 获取新对象上的主键键值 -- 允许动态变更修改键值
294        List<FieldAndColValue> priKeys = JpaAnnotationUtil.getIdFieldAndColumnValues(newBean);
295        // 获取主键键值，并赋予到
296        for(FieldAndColValue fieldAndColValue:priKeys){
297            updateMap.put(fieldAndColValue.getColumn().name().toLowerCase(),fieldAndColValue.getFieldValue());
298        }
299
300        return updateMap;
301    }
