001package com.edb.cloud.jfinal.activerecord.db.generator.util;
002
003import com.jfinal.kit.Kv;
004import com.jfinal.template.Engine;
005import com.jfinal.template.source.ClassPathSourceFactory;
006import lombok.extern.slf4j.Slf4j;
007import org.slf4j.Logger;
008import org.slf4j.LoggerFactory;
009
010import java.io.*;
011
012/**
013 * @program: xzw-coreframework-parent
014 * @description: Engine模板工具
015 * @author: MrYang
016 * @create: 2018-05-12 14:38
017 **/
018@Slf4j
019public class EngineUtil {
020
021
022    /**
023     * 根据具体模板生成文件
024     * @param templateFileName  模板文件名称
025     * @param kv                渲染参数
026     * @param filePath          输出目录
027     * @return
028     */
029    public static boolean render(String baseTemplatePath, String templateFileName, Kv kv, StringBuilder filePath)  {
030        //
031        // baseTemplatePath 暂时无用途
032        // 根据路径生成文件
033        return render(templateFileName,kv,filePath,false);
034    }
035
036    /**
037     * 不追加写入生成文件
038     * @param templateFileName
039     * @param kv
040     * @param filePath
041     * @return
042     */
043    public static boolean render(String templateFileName, Kv kv, StringBuilder filePath)  {
044        return render(templateFileName,kv,filePath,false);
045    }
046
047    /**
048     * 根据路径生成文件
049     * @param templateFileName
050     * @param kv
051     * @param filePath
052     * @return
053     */
054    public static boolean render(String templateFileName, Kv kv, StringBuilder filePath, boolean isAppend)  {
055        BufferedWriter output = null;
056        PrintWriter pw = null;
057        try {
058            log.debug("模板路径:{}",templateFileName);
059            log.debug("输出路径:{}",filePath);
060
061            File file = new File(filePath.toString());
062            File path = new File(file.getParent());
063            if ( ! path.exists() ) {
064                // 创建所有上级目录
065                path.mkdirs();
066            }
067
068            FileWriter fw = new FileWriter(file);
069
070            // 是否追加写入
071            if(isAppend){
072                // 追加写入的对象
073                pw = new PrintWriter(fw);
074                Engine.use()
075                        // 设置模板路径
076                        //.setBaseTemplatePath(baseTemplatePath)
077                        .setSourceFactory(new ClassPathSourceFactory())
078                        // 获取模板 - 通过指定文件名的方式获取 ,请勿用 baseTemplatePath 拼接，会无法获取到
079                        .getTemplate(templateFileName)
080                        // 键值对的方式写入文件里
081                        .render(kv, pw);
082            }else{
083                // 不追加写入
084                output = new BufferedWriter(fw);
085                Engine.use()
086                        // 设置模板路径
087                        //.setBaseTemplatePath(baseTemplatePath) ,请勿用 baseTemplatePath 拼接，会无法获取到
088                        .setSourceFactory(new ClassPathSourceFactory())
089                        // 获取模板 - 通过指定文件名的方式获取
090                        .getTemplate(templateFileName)
091                        // 键值对的方式写入文件里
092                        .render(kv, output);
093            }
094
095
096
097            return true;
098        } catch (IOException e) {
099            e.printStackTrace();
100            return false;
101        }finally{
102            try { if( output != null ) output.close(); } catch (IOException e) {}
103        }
104    }
105
106}
