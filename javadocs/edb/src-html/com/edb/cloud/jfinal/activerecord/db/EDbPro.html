092    /**
093     * 保存jpa对象
094     * @param m
095     * @param <M>
096     * @return
097     */
098    public <M> boolean save(M m){
099        if(m instanceof String){
100            throw new RuntimeException(" 直接执行sql请使用 EDb.update(sql) ");
101        }
102        Class<M> mClass = getRealJpaClass(m);
103        Table table = JpaAnnotationUtil.getTableAnnotation(mClass);
104        // 获取主键键值
105        String keys = JpaAnnotationUtil.getPriKeys(mClass);
106        // 获取所有字段列表
107        List<FieldAndColValue> coumns  = JpaAnnotationUtil.getCoumnValues(m);
108        Record record = new Record();
109        // 字段赋值
110        for(FieldAndColValue fieldAndColumn : coumns){
111            // 字段赋值
112            record.set(fieldAndColumn.getColumn().name().toLowerCase(), fieldAndColumn.getFieldValue());
113        }
114
115
116
117
118        // 保存前的监听
119        if(eDbListener!=null){
120            // 获取所有字段列表
121            List<FieldAndColumn> allCoumns  = JpaAnnotationUtil.getCoumns(mClass);
122            Map<String,Object> dataMap =  new CaseInsensitiveMap(record.getColumns());
123            // 执行对象方法
124            eDbListener.beforeSave(mClass,dataMap,allCoumns);
125            // 对象更新拷贝
126            BeanUtil.copyProperties(dataMap,record.getColumns());
127            // 释放map对象
128            dataMap = null;
129        }
130
131        // 保存前的方法事件
132        Method beforeUpdate = JpaAnnotationUtil.getMethod(mClass, EDbSave.class);
133        //
134        if(beforeUpdate != null){
135            // 替换成 忽略 大小写的 map
136            Map<String,Object> dataMap =  new CaseInsensitiveMap(record.getColumns());
137            // 执行对象方法
138            ReflectUtil.invoke(m, beforeUpdate, dataMap,coumns);
139            // 对象更新拷贝
140            BeanUtil.copyProperties(dataMap,record.getColumns());
141            // 释放map对象
142            dataMap = null;
143        }
144
145        // 判断数据库类型
146        if(this.getConfig().getDialect() instanceof PostgreSqlDialect) {
147            // 目前只做了mysql 和 pg 版本的兼容
148            for (FieldAndColValue fieldAndColumn : coumns) {
149                if (fieldAndColumn.getIsPriKey() && fieldAndColumn.getFieldValue() == null) {
150                    // 剔除字段
151                    record.remove(fieldAndColumn.getColumn().name().toLowerCase());
152                }
153            }
154        }
155
156
157        // 保存到数据库
158        boolean resultStatus = this.save(table.name(),keys,record);
159        // 由于结构会返回主键字段，所以可以反向赋值 -- 由于是基于jpa标准，字段可能非正常字段，赋值会引起可能性的错误或不一致
160        //BeanUtil.fillBeanWithMap(record.getColumns(),m,false);
161        // 获取主键字段
162        //coumns = JpaAnnotationUtil.getIdFieldAndColumnValues(m);
163        // 所有字段重新赋值，可能主键或者自定义键值产生变更（通过 beforeSave 方法调整变更的对象）
164        for(FieldAndColumn fieldAndColumn : coumns) {
165            // 字段赋值
166            ReflectUtil.setFieldValue(m,fieldAndColumn.getField(),record.get(fieldAndColumn.getColumn().name().toLowerCase()));
167        }
168
169        return resultStatus;
170    }

349    /**
350     * 使用 insert values(...),(...) 的方式批量插入对象
351     * @param objs
352     * @param batchSize
353     * @param <T>
354     */
355    public <T>  int  insertValues(Class<T> tClass,List<T> objs,int batchSize){
356        int valueSize = 0;
357        // 如果对象为null不管
358        if( objs == null || objs.size() == 0){
359            return 0;
360        }
361        // 获取表注解
362        Table table = AnnotationUtil.getAnnotation(tClass, Table.class);
363
364
365        // 表头部分
366        StringBuffer tableHead = new StringBuffer(100);
367        // 拼接sql -- 默认每1000为一批次提交
368        StringBuffer inserValues = new StringBuffer(batchSize * 1024);
369        // 设置表头名称
370        tableHead.append(" insert into ").append(table.name()).append("(");
371        // 获取所有字段列表
372        List<FieldAndColumn> coumns  = JpaAnnotationUtil.getCoumns(tClass);
373
374
375        for(FieldAndColumn fieldAndColumn:coumns){
376            // 拼接字段
377            tableHead.append(fieldAndColumn.getColumn().name()).append(",");
378        }
379        // 删除最后一个逗号
380        tableHead.deleteCharAt(tableHead.length() - 1);
381        // 闭环 -- insert into table(...)
382        tableHead.append(") values");
383        Object value = null;
384        T t = null;
385        // for 循环
386        for(int i=0;i<objs.size();i++){
387            // 如果对象有值，并且为一批次时，则进行一次提交
388            if(inserValues.length() > 0 && i % batchSize == 0){
389                // 裁剪末尾逗号
390                inserValues.deleteCharAt(inserValues.length() - 1);
391                // 执行语句
392                valueSize += this.update(tableHead.toString() + inserValues.toString());
393//                System.out.println(tableHead.toString() + inserValues.toString());
394                // 主动清空insert对象
395                inserValues.delete( 0, inserValues.length() );
396            }
397
398            // 赋予对象
399            t = objs.get(i);
400            // 拼接起始位置
401            inserValues.append("(");
402            // 赋予数据
403            for(FieldAndColumn fieldAndColumn:coumns){
404                // 数据项
405                value = ReflectUtil.getFieldValue(t,fieldAndColumn.getField());
406                // 只允许对象只有 String Inter Long BigDecimal 这几种常用类型 (便于扩展) ,枚举类型不建议，虽然也能实现，但是存在各种不确定性，请勿玩花样，避免代码太过复杂恶心
407                if(value instanceof String){
408                    inserValues.append("'").append(value).append("',");
409                }else
410                if(value instanceof Date){
411                    inserValues.append("'").append(DateUtil.formatDateTime((Date) value)).append("',");
412                }
413                else{
414                    // 如果是 pg
415                    if (value == null && this.getConfig().getDialect() instanceof PostgreSqlDialect && fieldAndColumn.getIsPriKey()){
416                        inserValues.append("nextval(pg_get_serial_sequence('").append(table.name().toLowerCase()).append("','").append(fieldAndColumn.getColumn().name().toLowerCase()).append("')),");
417                    }else{
418                        inserValues.append(value).append(",");
419                    }
420
421                }
422            }
423            // 裁剪末尾逗号
424            inserValues.deleteCharAt(inserValues.length() - 1);
425            // 拼接末尾
426            inserValues.append("),");
427        }
428
429
430        // 如果存在未执行的sql，则需要执行一次
431        if(inserValues.length() > 0 ){
432            // 裁剪末尾逗号
433            inserValues.deleteCharAt(inserValues.length() - 1);
434            // 执行语句
435            valueSize += this.update(tableHead.toString() + inserValues.toString());
436//            System.out.println(tableHead.toString() + inserValues.toString());
437            // 主动清空insert对象
438            inserValues.delete( 0, inserValues.length() );
439        }
440        return valueSize;
441    }

464     * @param mClass
465     * @param updateData
466     * @param <M>
467     * @return
468     */
469    public <M> boolean update(Class<M> mClass,Map<String,Object> updateData){
470        // 返回表对象 -- 便于获取表名称
471        Table table = JpaAnnotationUtil.getTableAnnotation(mClass);
472        // 获取主键键值 -- 小写
473        String keys = JpaAnnotationUtil.getPriKeys(mClass);
474        Record record = new Record();
475        // 变更对象 -- 获取到变更的数据库字段值,反向填充到map
476        Map<String,Object> dataMap = new HashMap<>();
477        // 必须有更新条件，所以不用判断null
478        if(updateData.size() > 0){
479            for(Map.Entry<String, Object> entry : updateData.entrySet()){
480                // 由于使用工具类取出来的数据库字段命名是小写，所以统一转小写 ，这点是因为你永远无法知道用户到底是小写还是大写的命名规则决定的
481                dataMap.put(entry.getKey().toLowerCase(),entry.getValue());
482            }
483        }
484
485        // 获取所有字段列表
486        List<FieldAndColumn> coumns  = JpaAnnotationUtil.getCoumns(mClass);
487
488        // 更新前的方法事件
489        Method beforeSave = JpaAnnotationUtil.getMethod(mClass, EDbUpdate.class);
490
491
492        Map<String,Object> updateDataMap = null;
493        // 保存前的监听
494        if(eDbListener!=null){
495            updateDataMap =  new CaseInsensitiveMap(dataMap);
496            // 执行对象方法
497            eDbListener.beforeUpdate(mClass,updateDataMap,coumns);
498            // 对象更新拷贝
499            BeanUtil.copyProperties(updateDataMap,dataMap);
500            // 释放map对象
501            updateDataMap = null;
502        }
503
504        //
505        if(beforeSave != null){
506            Object ojb = null;
507            try {
508                ojb = mClass.newInstance();
509            } catch (Throwable e) {
510                e.printStackTrace();
511            }
512            // 替换成 忽略 大小写的 map
513            updateDataMap =  new CaseInsensitiveMap(dataMap);
514            // 执行对象方法
515            ReflectUtil.invoke(ojb, beforeSave, updateDataMap,coumns);
516            // 对象更新拷贝
517            BeanUtil.copyProperties(updateDataMap,dataMap);
518            // 释放map对象
519            updateDataMap = null;
520        }
521        record.setColumns(dataMap);
522        // 更新对象
523        return this.update(table.name(),keys,record);
524    }
