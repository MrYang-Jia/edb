001package com.edb.cloud.jfinal.activerecord.db.listener;
002
003import com.edb.cloud.jfinal.activerecord.db.EDbPro;
004import com.edb.cloud.jfinal.activerecord.db.dto.EDBListenerResult;
005import com.edb.cloud.jfinal.activerecord.db.dto.FieldAndColumn;
006
007import java.util.List;
008import java.util.Map;
009
010/**
011 * jpa监听对象
012 * 1、保存前监听 -- @EDbSave 之前触发
013 * 2、更新前监听 -- @EDbUpdate 之前触发
014 */
015public interface EDbListener {
016
017    /**
018     * jpa对象统一保存前的实现逻辑
019     * @param jpaClass -- 表对象
020     * @param saveMap -- 保存前的对象数据
021     * @param coumns  -- 保存对象的所有字段信息
022     */
023    public void beforeSave(Class jpaClass,Map<String,Object> saveMap, List<FieldAndColumn> coumns);
024
025    /**
026     * jpa对象统一更新前的实现逻辑
027     * @param jpaClass -- 表对象
028     * @param updateMap  -- 更新前的对象数据
029     * @param coumns -- 更新对象的所有字段信息
030     */
031    public void beforeUpdate(Class jpaClass,Map<String,Object> updateMap, List<FieldAndColumn> coumns);
032
033    /**
034     * 执行删除前的相关操作
035     * @param eDbPro -- 操作的数据库对象
036     * @param jpaClass -- 表对象
037     * @param deleteMaps -- 指定需要删除的集合(默认只给予主键对应的id)
038     * @param coumns -- 表字段的相关信息
039     * @return 返回true才能继续执行删除，否则不做任何处理
040     */
041    public EDBListenerResult beforeDelete(EDbPro eDbPro,Class jpaClass,List<Map<String,Object>> deleteMaps, List<FieldAndColumn> coumns);
042
043}
