045    /**
046     * 表数据初始化
047     * @param table
048     * @param columnList
049     */
050    public static void loadTableData( GenTable table, List<GenTableColumn> columnList){
051
052        String tableName = table.getTableName().trim();
053
054        // 先获取表对象属性
055        Record tableRecord = null;
056        if(edbPro.getConfig().getDialect() instanceof PostgreSqlDialect){
057            tableRecord = edbPro.findFirst(GenPg.getTableInfoSql(tableName));
058        }else{
059            tableRecord = edbPro.findFirst(GenMysql.getTableInfoSql(tableName));
060        }
061
062        if(tableRecord==null){
063            throw new  RuntimeException(" 数据库不存在该表： "+tableName);
064        }
065
066        // 转成忽略大小写的map对象,便于取值
067        Map<String,Object> tableInfo = new CaseInsensitiveMap(tableRecord.getColumns() );
068
069        // 转换成bean对象
070        GenTable genTable = BeanUtil.mapToBean(tableInfo,GenTable.class,false);
071
072        if(table.getCreater() != null){
073            genTable.setCreater(table.getCreater());
074        }else{
075            // 回填创建人信息
076            table.setCreater(GenJdbc.creater);
077        }
078
079        if(table.getModelName() != null ){
080            genTable.setModelName(table.getModelName());
081        }
082
083        if(table.getProjectName() != null ){
084            genTable.setProjectName(table.getProjectName());
085        }
086
087        // 必须使用赋值的方式，否则对象传递无法发挥作用，对象内存变化无法指定
088        BeanUtils.copyProperties(genTable,table);
089
090
091        List<Record> columns = null;
092        if(edbPro.getConfig().getDialect() instanceof PostgreSqlDialect){
093            columns = edbPro.find(GenPg.getTableColumnsSql(tableName));
094        }else{
095            columns = edbPro.find(GenMysql.getTableColumnsSql(tableName));
096        }
097
098        GenTableColumn genTableColumn ;
099        String[] nums ;
100        // 指定类型
101        for(Record column:columns){
102            genTableColumn = new GenTableColumn();
103            genTableColumn = BeanUtil.mapToBean(column.getColumns(),GenTableColumn.class,false);
104            // 判断我方数据库字段
105            if(GenJdbc.filedTypeMap.get(genTableColumn.getDataType()) == null){
106                // 回填表字段名称
107                throw new RuntimeException("字段: "+genTableColumn.getColumnName()+" ;目前没有匹配的类型:" + genTableColumn.getDataType() + "建议使用对应的通配类型");
108            }
109            // 指定字段的java类型
110            genTableColumn.setColumnType(GenJdbc.filedTypeMap.get(genTableColumn.getDataType()));
111            // 驼峰式字段名
112            genTableColumn.setColumnCode(StrUtil.toCamelCase(genTableColumn.getColumnName().toLowerCase()));
113            // 通过sql语句获取的字段长度
114            if(genTableColumn.getMaxL()!=null && genTableColumn.getMaxL().contains(",")){
115                nums = genTableColumn.getMaxL().split(",");
116                // 长度
117                genTableColumn.setMaxL(nums[0]);
118                // 根据长度回填 N 个 9，代表最大值
119                genTableColumn.setMaxValue(String.join("", Collections.nCopies(Integer.valueOf(nums[0]) - Integer.valueOf(nums[1]), "9")));
120                // 小数位数
121                genTableColumn.setDecimalDigit(nums[1]);
122            }else{
123
124                if(genTableColumn.getMaxL().length()>0){
125                    // 根据长度回填 N 个 9，代表最大值
126                    genTableColumn.setMaxValue(String.join("", Collections.nCopies(Integer.valueOf(genTableColumn.getMaxL()), "9")));
127                }
128                genTableColumn.setDecimalDigit("0");
129            }
130
131            // 赋予对象
132            columnList.add(genTableColumn);
133        }
134
135
136        // 回填表主键
137        for(GenTableColumn column:columnList){
138            if(column.getColumnKey().equalsIgnoreCase("PRI")){
139                table.setPriKeyClassName(column.getColumnCode());
140                break;
141            }
142        }
143
144
145        // 填充className字段
146        loadClassName(table);
147    }

149    /**
150     * 装载className字段
151     * @param table
152     */
153    public static void loadClassName(GenTable table){
154        String preTableName = "";
155        String className = "";
156        String tableName = table.getTableName();
157        // 去掉第一个下划线表前缀，避免前端暴露表对象全名称
158        //tableName = tableName.substring(tableName.indexOf("_")+1,tableName.length());
159        // 非空的时候替换
160        if(StrUtil.isNotEmpty(GenJdbc.tablePreRemove)){
161            preTableName = GenJdbc.tablePreRemove.trim();
162            if(preTableName.indexOf("_") < 0){
163                preTableName +="_";
164            }
165            // 表前缀转小写
166            preTableName = preTableName.toLowerCase();
167            // 表名转小写并做替换处理
168            className = tableName.toLowerCase().replaceAll(preTableName,"");
169            // 驼峰转换
170            className = StrUtil.toCamelCase(className);
171            // 设置javaClass表名称
172            table.setClassName(className);
173        }else{
174            // 设置javaClass表名称, false 大驼峰
175            table.setClassName(StrUtil.upperFirst(StrUtil.toCamelCase(tableName)));
176
177        }
178        // 转小驼峰 - 大驼峰的首字母小写即可
179        table.setSmallClassName(StringUtils.uncapitalize(table.getClassName()));
180    }

183    /**
184     * 加载路径填充封装
185     * @param objPre
186     * @param projectUrl
187     * @param packageName
188     * @param table
189     */
190    public static void loadObjUrlAndObjPackageName(String objPre,String projectUrl,String packageName,GenTable table){
191        Map<String,Object> tableMap = BeanUtil.beanToMap(table);
192
193        // 去除空格
194        projectUrl = projectUrl.trim();
195        packageName = packageName.trim();
196
197        // 包名格式判断
198        if(packageName.indexOf(".")<0){
199            throw new  RuntimeException(" packageName格式有误，必须是com.xxx.xxx的格式 ; 当前格式： "+ packageName);
200        }
201
202        String packageUrl = "";
203        // 如果不是以文件符结束的
204        if(!projectUrl.endsWith(File.separator)){
205            projectUrl += File.separator;
206        }
207
208        //table.setEntityProjectUrl(projectUrl);
209        tableMap.put(objPre + "ProjectUrl",projectUrl);
210
211        // 替换成文件目录
212        packageUrl = packageName.replaceAll("\\.","\\\\");
213        // 如果不是以文件符结束的
214        if(!packageUrl.endsWith(File.separator)){
215            packageUrl += File.separator;
216        }
217        //table.setEntityPackageUrl(packageUrl);
218        tableMap.put(objPre + "PackageUrl",packageUrl);
219        //去除最后一个空格
220        if(packageName.endsWith(".")){
221            packageName = packageName.substring(0,packageName.length()-1);
222        }
223        //table.setEntityPackageName(packageName);
224        tableMap.put(objPre + "PackageName",packageName);
225        //
226        BeanUtils.copyProperties(BeanUtil.mapToBean(tableMap,table.getClass(),false),table);
227    }

246     * 生成实体类
247     * @param projectUrl -- 实体对象目录
248     * @param packageName -- 实体对象包名
249     * @param genTable -- 实体表对象
250     * @param genTableColumns -- 实体表对象字段集
251     */
252    public static void generatorBaseJpaBean(String projectUrl,String packageName,GenTable genTable, List<GenTableColumn> genTableColumns){
253
254        // 加载文件路径 -- 实体对象目录,实体的包名
255        loadEntityUrlAndPackageName(projectUrl,packageName,genTable);
256        // 模板的位置
257        URL baseUrl = EDbGenCode.class.getResource("/edb-template");
258
259        // 生成的java文件名
260        String javaFileName = "Base" + genTable.getClassName()+".java" ;
261
262        StringBuilder outJavaFile = new StringBuilder()
263                // .append(System.getProperty("user.dir"))
264                // 实体对象的目录
265                .append(genTable.getEntityProjectUrl() )
266                // 大驼峰
267                .append(javaFileName);
268        File file = new File(outJavaFile.toString());
269
270        if(file.exists()){
271            System.out.println(" === "+ javaFileName +" 文件重写 === ");
272            //file.delete();
273        }
274
275
276        // 指向目录
277        EngineUtil.render(baseUrl.getPath(),
278                // 单独指定，靠上面的传入值只会获取编译后的包的路径，无法取到正确的路径信息
279                "/edb-template/java/baseJpa.tpl",
280                Kv.by("genClass", genTable)
281                        .set("fields",genTableColumns)
282                        .set("nowdatetime", DateUtil.now())
283                ,outJavaFile
284        );
285    }

541    /**
542     * 获取基础路径
543     * @param copyWindowsPath
544     * @return
545     */
546    public static String getBaseUrl(String copyWindowsPath){
547        // 由于 \ 字符串转义不好处理，所以需要先转 - 最后再转 \
548        String currentProject = System.getProperty("user.dir");
549        currentProject = currentProject.replaceAll("\\\\","-");
550        copyWindowsPath = copyWindowsPath.replaceAll("\\\\","-");
551        copyWindowsPath = copyWindowsPath.replaceAll(currentProject,"");
552        return  copyWindowsPath.replaceAll("-","\\\\");
553    }

555    /**
556     * 将基础路径转换成包名
557     * @param copyWindowsPath
558     * @return
559     */
560    public static String getBasePackageName(String copyWindowsPath){
561        // 由于 \ 字符串转义不好处理，所以需要先转 - 最后再转 \
562        String currentProject = System.getProperty("user.dir");
563        currentProject = currentProject.replaceAll("\\\\","-");
564        copyWindowsPath = copyWindowsPath.replaceAll("\\\\","-");
565        String replaceSrcStr = "-src-main-java-";
566        int idx = copyWindowsPath.indexOf(replaceSrcStr) + replaceSrcStr.length();
567        copyWindowsPath = copyWindowsPath.substring(idx,copyWindowsPath.length());
568        return  copyWindowsPath.replaceAll("-","\\.");
569    }
