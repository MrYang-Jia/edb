001package com.edb.cloud.jfinal.activerecord.db.annotation;
002
003import java.lang.annotation.ElementType;
004import java.lang.annotation.Retention;
005import java.lang.annotation.RetentionPolicy;
006import java.lang.annotation.Target;
007
008/**
009 * 数据库关系对象获取
010 */
011@Target({ ElementType.METHOD , ElementType.FIELD})
012@Retention(RetentionPolicy.RUNTIME)
013public @interface EDbRel {
014
015    // relKey -- 指定 relKey 可用于多对象时取自定义的 relKey 关联对象信息
016    String relKey() default "";
017
018    // 指定关联字段
019    String[] relColumn() default {};
020
021    // 拼接其他字符串 ，以 and 或 or 开始拼接
022    String appendSql() default "";
023
024    // 默认10个，避免关系数量太多，导致获取的列表数据
025    int limit() default 10;
026
027    // 有特殊需求可以指定获取的位置
028    int offset() default 0;
029
030
031}
