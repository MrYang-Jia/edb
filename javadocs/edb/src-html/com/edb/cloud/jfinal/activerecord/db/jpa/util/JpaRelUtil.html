042    /**
043     * 获取关系对象返回的结果 或 返回的 sqlPara 对象 (这个设计起始挺不好的，会存在两种可能性的返回值)
044     * @param relKey -- 如果存在多个同一对象，用key做区分，否则无法正确赋予对象
045     * @param fields -- 字段
046     * @param limit -- 最多返回多少条记录
047     * @param offset -- 从第几条记录开始返回，起始值为0
048     * @param eDbPro -- 数据源
049     * @param oriJpa -- 原始jpa对象
050     * @param typeName -- class 的全名称 ，便于加载生成 class 对象 或 做类比较
051     * @param isReturnResutl true - 返回数据对象结果，false 返回 Map<String, List<SqlPara>>
052     * @param isFutrue 是否异步回调, isReturnResutl = true 时有效， 并且返回的结果集变更为 List<Future<Object>>
053     * @return
054     */
055    public static Object getRelObject(String relKey,
056                               String fields,
057                               Integer limit,
058                               Integer offset,
059                               EDbPro eDbPro,
060                               Object oriJpa,
061                               String typeName,
062                               Method method,
063                               boolean isReturnResutl,
064                               boolean isFutrue){
065        // 获取所有字段 -- 包含
066        List<FieldAndRel> fieldRels = JpaAnnotationUtil.getFieldRels(oriJpa.getClass());
067
068        // 字段的所有查询值
069        List<FieldAndColValue> fieldAndColValues =  null;
070        //
071        EDbRel eDbRel = null;
072        Type fieldType = null;
073        Class<?> entityClass = null;
074        Map<String,Object> tableData = null;
075        //
076        SqlPara sqlPara = null;
077        //
078        Map<String, List<SqlPara>> sqlParasMap = null;
079        if(!isReturnResutl){
080            sqlParasMap = new HashMap<>();
081        }
082
083        //
084        Object object = null;
085
086        JpaRelTask jpaRelTask = null;
087        Future<Object> futureRate = null;
088        // 异步结果集
089        List<Future<Object>> resultFutrues = null;
090        if(isFutrue){
091            resultFutrues = new ArrayList<>();
092        }
093
094
095//        System.out.println("返回的方法类型：" + returnType.getTypeName());
096        // 有关联的rel注解才进行扩展
097        if(fieldRels != null){
098
099            for(FieldAndRel fieldAndRel :fieldRels){
100                // -- 判断指向的唯一标识是否有值
101                if(!StringUtils.isEmpty(relKey) ){
102                    // 如果 relKey 有传递值，则必须匹配上对应的 relKey 字段才进行数据返回操作，否则跳过
103                    if (!relKey.equals(fieldAndRel.getEDbRel().relKey())){
104                        continue;
105                    }
106                }
107                // 字段类型
108                fieldType = fieldAndRel.getField().getGenericType();
109                // 返回对象 和 rel 注解匹配，则返回对应的结果集 或者是有指向唯一标识 relKey
110                if(relKey!=null || typeName == null || fieldType.getTypeName().equals(typeName)){
111
112                    if(StringUtils.isEmpty(relKey) ){
113                        //
114//                        if(fieldAndRel.getEDbRel().relKey().length() > 0 && typeName!=null){
115//                            throw new RuntimeException(" 请使用 edbPro.getRelKey 返回指定对象的结果 或 edbPro.getAllRel() 返回结果集 ");
116//                        }
117                        if(typeName != null){
118                            // 如果方面名和字段名提供的方法名不一致，则不是所命中对象的get方法
119                            if(method!=null && !method.getName().equals(JpaAnnotationUtil.getFieldMethod(fieldAndRel.getField(),oriJpa.getClass()).getName())){
120                                continue;
121                            }
122                        }
123
124                    }
125
126                    // 有
127                    if(fieldAndColValues == null){
128                        fieldAndColValues =  JpaAnnotationUtil.getCoumnValues(oriJpa);
129                    }
130                    // 某个关联字段
131                    eDbRel = fieldAndRel.getEDbRel();
132                    // 表结构
133                    tableData = new HashMap<>();
134                    // 判断是否是 ParameterizedType 对象
135                    if(fieldType instanceof ParameterizedType){
136                        // List
137                        entityClass = (Class<?>)((ParameterizedType) fieldType).getActualTypeArguments()[0];
138                        // 按默认配置来取条数
139                        sqlPara = getSqlPara(fields,limit,offset,eDbPro,fieldAndColValues,entityClass,eDbRel,tableData);
140                        // 是否返回结果集
141                        if(isReturnResutl){
142                            if(isFutrue){
143                                jpaRelTask = new JpaRelTask(oriJpa,fieldAndRel.getField(),entityClass,sqlPara,eDbPro,true);
144                                //  执行
145                                futureRate = EDb.edbFutruePools.get(eDbPro.getConfig().getName()).submit(jpaRelTask);
146                                resultFutrues.add(futureRate);
147                            }else{
148                                object = eDbPro.find(entityClass,sqlPara);
149                                // 字段赋值 -- 反射赋值会比较消耗毫秒数
150                                ReflectUtil.setFieldValue(oriJpa,fieldAndRel.getField(),object);
151
152
153                            }
154
155                            if( relKey !=null || typeName != null){
156                                // 如果是有异步指定，则必须获得指定结果后再退出循环
157                                if(relKey !=null && isFutrue && resultFutrues.size()>0){
158                                    return resultFutrues;
159                                }else{
160                                    return object;
161                                }
162                            }
163
164
165                        }else{
166                            // 不返回结果集
167                            if( sqlParasMap.get(fieldType.getTypeName()) == null){
168                                sqlParasMap.put(fieldType.getTypeName(),new ArrayList<SqlPara>());
169                            }
170                            sqlParasMap.get(fieldType.getTypeName()).add(sqlPara);
171                        }
172                    }else{
173                        // 单体对象
174                        try {
175                            entityClass = Class.forName(fieldType.getTypeName());
176                        } catch (ClassNotFoundException e1) {
177                            e1.printStackTrace();
178                        }
179                        // 返回sql对象 -- 参数单独返回，最后统一拼凑
180                        sqlPara = getSqlPara(fields,1,0,eDbPro,fieldAndColValues,entityClass,eDbRel,tableData);
181                        if(isReturnResutl) {
182                            if(isFutrue){
183                                jpaRelTask = new JpaRelTask(oriJpa,fieldAndRel.getField(),entityClass,sqlPara,eDbPro,false);
184                                //  执行
185                                futureRate = EDb.edbFutruePools.get(eDbPro.getConfig().getName()).submit(jpaRelTask);
186                                resultFutrues.add(futureRate);
187                            }else{
188                                object = eDbPro.findFirst(entityClass, sqlPara);
189                                // 字段赋值 -- 反射赋值会比较消耗毫秒数
190                                ReflectUtil.setFieldValue(oriJpa, fieldAndRel.getField(), object);
191                            }
192                            if( relKey!=null || typeName != null){
193                                // 如果是有异步指定，则必须获得指定结果后再退出循环
194                                if(relKey !=null && isFutrue && resultFutrues.size()>0){
195                                    return resultFutrues;
196                                }else{
197                                    return object;
198                                }
199                            }
200                        }else{
201                            if( sqlParasMap.get(fieldType.getTypeName()) == null){
202                                sqlParasMap.put(fieldType.getTypeName(),new ArrayList<SqlPara>());
203                            }
204                            sqlParasMap.get(fieldType.getTypeName()).add(sqlPara);
205                        }
206                    }
207                }
208            }
209        }
210
211        //
212        if(sqlParasMap!=null && sqlParasMap.size() > 0){
213            return sqlParasMap;
214        }
215
216        //
217        if(isFutrue && resultFutrues.size()>0){
218            return resultFutrues;
219        }
220
221
222        return object;
223    }

227    /**
228     * 获取sql对象 -- 需要开启enjoy-sql模板  sqlTplList.add("/edb/sql/all.sql");
229     * @param fields
230     * @param limit
231     * @param offset
232     * @param eDbPro
233     * @param fieldAndColValues
234     * @param entityClass
235     * @param eDbRel
236     * @param tableData
237     * @param columns
238     * @return
239     */
240    @Deprecated
241    public static SqlPara getSqlPara(String fields,
242                              Integer limit,
243                              Integer offset,
244                              EDbPro eDbPro,
245                              List<FieldAndColValue> fieldAndColValues,Class<?> entityClass,
246                              EDbRel eDbRel,
247                              Map<String,Object> tableData,
248                              List<String> columns){
249        // 获取jpa对象的 table 注解
250        Table table = JpaAnnotationUtil.getTableAnnotation(entityClass);
251        SqlPara sqlPara = new SqlPara();
252
253        // 自定义查询字段 -- 估计用到的场合少，但是也有特殊情况需要有，甚至可以针对特别字段自己做一些函数表达式等处理
254        if(fields != null){
255            tableData.put(JpaRelPip.fields,fields);
256        }
257
258        // 赋予表名称
259        tableData.put(JpaRelPip.tableName,table.name());
260
261        for (String columnName : eDbRel.relColumn()){
262            // 匹配
263            for(FieldAndColValue fieldAndColValue : fieldAndColValues){
264                // 匹配数据库字段
265                if(fieldAndColValue.getColumn().name().toLowerCase().equals(columnName.toLowerCase())){
266                    // 表字段 -- 暂无用处
267                    columns.add(columnName);
268                    // 入参赋值
269                    sqlPara.addPara(fieldAndColValue.getFieldValue());
270                }
271            }
272        }
273        // where条件
274        tableData.put(JpaRelPip.params,columns);
275        // 拼接的sql
276        String appendSql = eDbRel.appendSql();
277        //
278        List<String> results =  ReUtil.findAll("(#\\(){1}(.*?)(\\){1})",appendSql,0);
279        // 字段名称
280        String columnName = "";
281        //
282        for(String regStr:results) {
283            // 获取字段 ，将匹配的字段前后部分替换成空字符串
284            columnName = regStr.replaceAll(jpaRelReg,"$1").replaceAll(" ","");
285            for(FieldAndColValue fieldAndColValue : fieldAndColValues){
286                if(fieldAndColValue.getColumn().name().toLowerCase().equals(columnName.toLowerCase())){
287                    // 入参赋值
288                    sqlPara.addPara(fieldAndColValue.getFieldValue());
289                }
290            }
291        }
292        // 统一替换成 ？
293        appendSql = appendSql.replaceAll(jpaRelReg,"?");
294
295        tableData.put(JpaRelPip.appendSql,appendSql);
296
297        // 是否有传入每次返回的最大条数
298        if(limit == null){
299            // 返回的数量
300            tableData.put(JpaRelPip.limit,eDbRel.limit());
301        }else{
302            tableData.put(JpaRelPip.limit,limit);
303        }
304
305        // 是否有传入自定义查询起始位
306        if(offset == null){
307            // 读取数据的位置
308            tableData.put(JpaRelPip.offset,eDbRel.offset());
309        }else{
310            // 读取数据的位置
311            tableData.put(JpaRelPip.offset,offset);
312        }
313
314        // sql对象 -- 因为还要指定模板加载，对于构建组件来说步骤太麻烦，所以去除
315        sqlPara.setSql(eDbPro.getSqlPara(EDbRelProxy.jpaEdbRelKey,tableData).getSql());
316        return sqlPara;
317    }

320    /**
321     * 获取sql对象
322     * @param fields
323     * @param limit
324     * @param offset
325     * @param eDbPro
326     * @param fieldAndColValues
327     * @param entityClass
328     * @param eDbRel
329     * @param tableData
330     * @return
331     */
332    public static SqlPara getSqlPara(String fields,
333                                     Integer limit,
334                                     Integer offset,
335                                     EDbPro eDbPro,
336                                     List<FieldAndColValue> fieldAndColValues,Class<?> entityClass,
337                                     EDbRel eDbRel,
338                                     Map<String,Object> tableData){
339        // 获取jpa对象的 table 注解
340        Table table = JpaAnnotationUtil.getTableAnnotation(entityClass);
341        StringBuffer sqlBuf = new StringBuffer(" select ");
342        SqlPara sqlPara = new SqlPara();
343        // 自定义查询字段 -- 估计用到的场合少，但是也有特殊情况需要有，甚至可以针对特别字段自己做一些函数表达式等处理
344        if(fields != null){
345            sqlBuf.append(" fields ");
346        }
347        else{
348            sqlBuf.append(" * ");
349        }
350        // 赋予表名
351        sqlBuf.append(" from ").append(table.name()).append(" where 1=1 ");
352
353        for (String columnName : eDbRel.relColumn()){
354            // 匹配
355            for(FieldAndColValue fieldAndColValue : fieldAndColValues){
356                // 匹配数据库字段
357                if(fieldAndColValue.getColumn().name().toLowerCase().equals(columnName.toLowerCase())){
358                    // 拼接sql
359                    sqlBuf.append(" and ").append(columnName).append(" = ? ");
360                    // 入参赋值
361                    sqlPara.addPara(fieldAndColValue.getFieldValue());
362                }
363            }
364        }
365        // where条件
366        // 拼接的sql
367        String appendSql = eDbRel.appendSql();
368        //
369        List<String> results =  ReUtil.findAll("(#\\(){1}(.*?)(\\){1})",appendSql,0);
370        // 字段名称
371        String fieldColumnName = "";
372        //
373        for(String regStr:results) {
374            // 获取字段 ，将匹配的字段前后部分替换成空字符串
375            fieldColumnName = regStr.replaceAll(jpaRelReg,"$1").replaceAll(" ","");
376            for(FieldAndColValue fieldAndColValue : fieldAndColValues){
377                // 原来是获取到数据库字段名去做替换，但是含义和代表较弱，所以替换成操作对象的key值
378//                if(fieldAndColValue.getColumn().name().toLowerCase().equals(columnName.toLowerCase())){
379//                    // 入参赋值
380//                    sqlPara.addPara(fieldAndColValue.getFieldValue());
381//                }
382                // 字段名
383                if(fieldAndColValue.getField().getName().equals(fieldColumnName)){
384                    // 入参赋值
385                    sqlPara.addPara(fieldAndColValue.getFieldValue());
386                }
387            }
388        }
389        // 统一替换成 ？
390        appendSql = appendSql.replaceAll(jpaRelReg,"?");
391        //  拼接的sql
392        sqlBuf.append(appendSql);
393        // 是否有传入每次返回的最大条数
394        if(limit == null){
395            // 返回的数量
396            sqlBuf.append(" limit ").append(eDbRel.limit());
397        }else{
398            sqlBuf.append(" limit ").append(limit);
399        }
400
401        // 是否有传入自定义查询起始位
402        if(offset == null){
403            // 读取数据的位置
404            sqlBuf.append(" offset ").append(eDbRel.offset());
405        }else{
406            // 读取数据的位置
407            sqlBuf.append(" offset ").append(offset);
408        }
409        // sql对象 -- 因为还要指定模板加载，对于构建组件来说步骤太麻烦，所以去除
410        sqlPara.setSql(sqlBuf.toString());
411        return sqlPara;
412    }
