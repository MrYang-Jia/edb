001package com.edb.cloud.jfinal.activerecord.db.proxy;

003
004import com.edb.cloud.jfinal.activerecord.db.EDbPro;
005import com.edb.cloud.jfinal.activerecord.db.jpa.util.JpaRelUtil;
006import lombok.Setter;
007import net.sf.cglib.proxy.Enhancer;
008import net.sf.cglib.proxy.MethodInterceptor;
009import net.sf.cglib.proxy.MethodProxy;
010import java.lang.reflect.Method;
011import java.lang.reflect.Type;
012
013/**
014 * @ClassName EDbRelProxy
015 * @Description: Jpa关系对象
016 * @Author 杨志佳
017 * @Date 2020/11/18
018 * @Version V1.0
019 **/
020public class EDbRelProxy implements MethodInterceptor {
021
022    public static String jpaEdbRelKey = "edb.EDbRel";
023
024    //要代理的原始对象
025    private Object oriJpa ;
026
027    // 自定义字段
028    @Setter
029    private String fields;
030
031    @Setter
032    private Integer limit;
033
034    @Setter
035    private Integer offset;
036
037    // 数据对象
038    private EDbPro eDbPro;
039
040
041
042    // 1- 创建代理对象
043    public <T> T createProcy(T target,EDbPro eDbPro){
044        // 需要判断是否已经是代理对象，如果是的话，无须二次代理
045//        System.out.println("需要被代理的类："+this.getClass().getSimpleName());
046        // 赋予操作对象
047        this.oriJpa = target;
048        // 赋予数据库对象
049        this.eDbPro = eDbPro;
050        //1-Enhancer类是CGLib中的一个字节码增强器，它可以方便的对你想要处理的类进行扩展
051        Enhancer enhancer=new Enhancer();
052        //2-将被代理类HuGe设置成父类
053        enhancer.setSuperclass(this.oriJpa.getClass());
054        //3-设置拦截器
055        enhancer.setCallback(this);
056        //4-动态生成一个代理类
057        Object objProxy = enhancer.create();
058        return (T) objProxy;
059    }
060
061    //2-实现MethodInterceptor的intercept方法
062    public Object intercept(Object obj, Method method, Object[] args, MethodProxy proxy) throws Throwable {
063//        System.out.println("before: " + method);
064        //调用proxy.invoke()方法，会报java.lang.StackOverflowError错误，原因是invoke()内部会一直被反复调用
065        //Object object = proxy.invoke(obj, args);
066//        Object object = proxy.invokeSuper(obj, args);
067        Object object = null;
068        // 获取调用方法的返回对象
069        Type returnType = method.getAnnotatedReturnType().getType();
070        // 如果返回类型有值，才进行关系扩展
071        if(returnType!= null ){
072            // 同步的方式，获取指定对象
073            object = JpaRelUtil.getRelObject(null,fields,limit,offset,eDbPro,oriJpa,returnType.getTypeName(),method,true,false);
074        }
075
076//        returnType.getTypeName()
077        // 如果不是rel对象，则返回原属性方法值
078        if(object == null){
079            object = proxy.invokeSuper(obj, args);
080        }
081        return object;
082    }

086}
