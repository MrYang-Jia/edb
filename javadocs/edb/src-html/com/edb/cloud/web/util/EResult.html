001package com.edb.cloud.web.util;
002
003import lombok.Data;
004
005import java.io.Serializable;
006import java.util.HashMap;
007import java.util.Map;
008
009/**
010 * 返回结果集对象 -- todo:待做一个新的项时，改用该api
011 */
012@Data
013public class EResult  implements Serializable {
014    // 0 成功，其他则为失败
015    private String code;
016    private String msg;
017    // 返回结果集
018    private Map<String,Object> data = new HashMap<>();
019    //总条数--分页条件此参数有值
020    private Long count;
021
022    /**
023     * 返回结果
024     * @param data
025     * @return
026     */
027    public static EResult succ(Object data) {
028        return succ("main",data);
029    }
030
031    /**
032     * 返回成功的结果信息
033     * @param key
034     * @param data
035     * @return
036     */
037    public static EResult succ(String key,Object data) {
038        EResult m = new EResult();
039        m.setCode("0");
040        m.getData().put(key,data);
041        m.setMsg("操作成功");
042        return m;
043    }
044
045    /**
046     * 返回成功的结果
047     * @param key
048     * @param mess
049     * @param data
050     * @return
051     */
052    public static EResult succ(String key,String mess, Object data) {
053        EResult m = new EResult();
054        m.setCode("0");
055        m.getData().put(key,data);
056        m.setMsg(mess);
057        return m;
058    }
059
060    /**
061     * 返回失败的信息
062     * @param mess
063     * @return
064     */
065    public static EResult fail(String mess) {
066        EResult m = new EResult();
067        m.setCode("-1");
068        m.setData(null);
069        m.setMsg(mess);
070        return m;
071    }
072
073    /**
074     * 返回失败的信息
075     * @param key
076     * @param mess
077     * @param data
078     * @return
079     */
080    public static EResult fail(String key,String mess, Object data) {
081        EResult m = new EResult();
082        m.setCode("-1");
083        m.getData().put(key,data);
084        m.setMsg(mess);
085        return m;
086    }
087}
