001package com.edb.cloud.util.json;
002
003
004import com.fasterxml.jackson.annotation.JsonInclude;
005import com.fasterxml.jackson.core.JsonParser;
006import com.fasterxml.jackson.core.JsonProcessingException;
007import com.fasterxml.jackson.databind.DeserializationFeature;
008import com.fasterxml.jackson.databind.JavaType;
009import com.fasterxml.jackson.databind.ObjectMapper;
010import com.fasterxml.jackson.databind.SerializationFeature;
011import com.fasterxml.jackson.databind.util.JSONPObject;
012import org.springframework.util.StringUtils;
013
014import java.io.IOException;
015import java.util.Collection;
016import java.util.List;
017import java.util.Map;
018import java.util.TimeZone;
019
020/**
021 * @ClassName EDbJsonUtil
022 * @Description: json 工具集
023 * @Author 杨志佳
024 * @Date 2020/10/22
025 * @Version V1.0
026 **/
027public class EDbJsonUtil {
028
029    public static ObjectMapper mapper;
030
031
032    static {
033        if(mapper==null){
034            mapper = new ObjectMapper();
035            mapper.configure(JsonParser.Feature.ALLOW_UNQUOTED_CONTROL_CHARS, true) ;
036            mapper.configure(JsonParser.Feature.ALLOW_BACKSLASH_ESCAPING_ANY_CHARACTER, true);
037            // 如果json中有新增的字段并且是实体类类中不存在的，不报错
038            mapper.configure(DeserializationFeature.FAIL_ON_UNKNOWN_PROPERTIES, false);
039            mapper.configure(SerializationFeature.WRITE_BIGDECIMAL_AS_PLAIN, true);
040            mapper.setTimeZone(TimeZone.getTimeZone("Asia/Shanghai"));
041            mapper.setSerializationInclusion(JsonInclude.Include.ALWAYS);
042            // 转换为格式化的json
043//            mapper.enable(SerializationFeature.INDENT_OUTPUT);
044        }
045    }
046
047    /**
048     * 转json
049     * @param object
050     * @return
051     */
052    public static String toJson(Object object) {
053
054        try {
055            return mapper.writeValueAsString(object);
056        } catch (IOException e) {
057
058            throw  new RuntimeException("write to json string error",e);
059        }
060    }
061
062    /**
063     * 格式化输出
064     * @param object
065     * @return
066     * @throws
067     */
068    public static String toJsonForFormat(Object object)  {
069
070        try {
071            return mapper.writerWithDefaultPrettyPrinter().writeValueAsString(object);
072        } catch (IOException e) {
073
074            throw  new RuntimeException("write to json string error",e);
075        }
076    }
077
078    /**
079     *
080     * @param jsonString
081     * @param clazz
082     * @param <T>
083     * @return
084     */
085    public static  <T> T fromJson(String jsonString, Class<T> clazz) {
086        if (StringUtils.isEmpty(jsonString)) {
087            throw  new RuntimeException("格式JSON不能为空");
088        }
089
090        try {
091            return mapper.readValue(jsonString, clazz);
092        } catch (IOException e) {
093            throw  new RuntimeException("parse json string error:",e);
094        }
095    }
096
097    /**
098     *
099     * @param jsonString
100     * @param javaType
101     * @param <T>
102     * @return
103     */
104    public static  <T> T fromJson(String jsonString, JavaType javaType) {
105        if (StringUtils.isEmpty(jsonString)) {
106            throw  new RuntimeException("格式JSON不能为空");
107        }
108        try {
109            return (T) mapper.readValue(jsonString, javaType);
110        } catch (IOException e) {
111            throw  new RuntimeException("parse json string error:",e);
112        }
113    }
114
115    public static JavaType contructCollectionType(Class<? extends Collection> collectionClass, Class<?> elementClass) {
116        return mapper.getTypeFactory().constructCollectionType(collectionClass, elementClass);
117    }
118
119
120    public static JavaType contructMapType(Class<? extends Map> mapClass, Class<?> keyClass, Class<?> valueClass) {
121        return mapper.getTypeFactory().constructMapType(mapClass, keyClass, valueClass);
122    }
123
124    public void update(String jsonString, Object object){
125        try {
126            mapper.readerForUpdating(object).readValue(jsonString);
127        } catch (JsonProcessingException e) {
128
129            throw  new RuntimeException("update json string:" + jsonString + " to object:" + object + " error.",e);
130        } catch (IOException e) {
131            throw  new RuntimeException("update json string:" + jsonString + " to object:" + object + " error.",e);
132        }
133    }
134
135
136    public static String toJsonP(String functionName, Object object) {
137        return toJson(new JSONPObject(functionName, object));
138    }
139
140    public static void enableEnumUseToString() {
141        mapper.enable(SerializationFeature.WRITE_ENUMS_USING_TO_STRING);
142        mapper.enable(DeserializationFeature.READ_ENUMS_USING_TO_STRING);
143    }
144
145    /**
146     * jsonList字符串转对象数组
147     * @param jsonList
148     * @param elementClass
149     * @param <T>
150     * @return
151     */
152    public static  <T> T parseArray(String jsonList, Class<?> elementClass) {
153        JavaType javaType = contructCollectionType(List.class, elementClass);
154        return fromJson(jsonList, javaType);
155    }
156
157
158}
