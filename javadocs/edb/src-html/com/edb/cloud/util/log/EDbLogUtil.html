001package com.edb.cloud.util.log;
002
003import ch.qos.logback.classic.LoggerContext;
004import ch.qos.logback.classic.joran.JoranConfigurator;
005import ch.qos.logback.core.joran.spi.JoranException;
006import ch.qos.logback.core.util.StatusPrinter;
007import lombok.extern.slf4j.Slf4j;
008import org.slf4j.LoggerFactory;
009
010/**
011 * @ClassName EDbLogUtil
012 * @Description: 日志工具类
013 * @Author 杨志佳
014 * @Date 2020/10/22
015 * @Version V1.0
016 **/
017@Slf4j
018public class EDbLogUtil {
019
020    /**
021     * log4J日志配置加载初始化
022     * 基础路径默认: src/main/resources/
023     */
024    public static void loadResoucesForSlf4j(String xmlUrl){
025        LoggerContext lc = (LoggerContext) LoggerFactory.getILoggerFactory();
026        JoranConfigurator configurator = new JoranConfigurator();
027        configurator.setContext(lc);
028        lc.reset();
029        try {
030            configurator.doConfigure("src/main/resources/"+xmlUrl);
031        } catch (JoranException e) {
032            e.printStackTrace();
033        }
034        StatusPrinter.printInCaseOfErrorsOrWarnings(lc);
035        log.debug(" sjf4j 日志装载成功 ");
036    }
037
038}
